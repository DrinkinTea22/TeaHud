event HudMedicCharged
{
	Animate	ChargeLabel 			FgColor		"Ubercharge Full"		Linear 0.0 0.1
	Animate	IndividualChargesLabel 	FgColor		"Ubercharge Full"		Linear 0.0 0.1

	RunEvent HudMedicChargedLoop	0.6
}

// call to loop HudHealthBonusPulse
event HudMedicChargedLoop
{
	RunEvent HudMedicCharged 0.0
}

event HudMedicChargedStop
{
	StopEvent HudMedicCharged 0.0
	StopEvent HudMedicChargedLoop 0.0
	
	Animate	ChargeLabel 			FgColor		"White"		Linear 0.0 0.1	
	Animate	IndividualChargesLabel 	FgColor		"White"		Linear 0.0 0.1
}
